class Q_NetworkAdapter(nn.Module):
    """
    Adapts the transformer Q_network for SB3.
    
    SB3 expects: forward(concatenated [obs_features, action]) → scalar Q-value
    Our Q_network expects: forward(state, action) with proper tensor shapes
    """
    
    def __init__(self,
                 state_dim=14, 
                 action_dim=8,
                 max_planets=40,
                 max_fleets=100, 
                 d_model=128, 
                 nhead=4, 
                 num_layers=3,  
                 dim_feedforward=2048, 
                 dropout=0.1):
        
        super().__init__()
        self.max_planets = max_planets
        self.max_fleets = max_fleets
        self.state_dim = state_dim
        self.action_dim = action_dim
        self.d_model = d_model
        self.num_state_tokens = state_dim // d_model
        
        # Use the actual Q_network from SAC.py
        self.q_net = Q_network(state_dim=state_dim, action_dim=action_dim,
                               max_planets=max_planets, max_fleets=max_fleets,
                               d_model=d_model, nhead=nhead, num_encoder_layers=num_layers,
                               dim_feedforward=dim_feedforward, dropout=dropout)
        
    
    def forward(self, state_action: Tensor) -> Tensor:
        # max_state * state_dim + max_action * action_dim = state_action_dim

        batch_size = state_action.shape[0]
        max_state_tokens = self.max_planets + self.max_fleets
        max_action_tokens = self.max_planets 
        state_action_dim = state_action.shape[1]
        state_flat = state_action[:, :(max_state_tokens * self.state_dim)]
        action_flat = state_action[:, (max_state_tokens * self.state_dim):]
        
        # Reshape to [batch_size, num_tokens, d_model]
        state = state_flat.view(batch_size, max_state_tokens, self.d_model)
        num_action_tokens = self.action_dim // self.d_model
        action = action_flat.view(batch_size, num_action_tokens, self.d_model)
        
        # Forward through Q_network (now handles batching natively)
        q_values = self.q_net(state, action)
        
        return q_values


class P_NetworkAdapter(nn.Module):
    """
    Adapts the transformer P_network (policy) for SB3.
    
    SB3 expects: latent_pi to output features, then mu/log_std heads process them
    Our P_network returns: (mu, sigma) directly
    """
    
    def __init__(self, 
                 d_model=128, 
                 state_dim=14,
                 action_dim=8,
                 max_planets=40,
                 max_fleets=100, 
                 nhead=4, 
                 num_layers=3,  
                 dim_feedforward=128, 
                 dropout=0.1):
        
        super().__init__()
        self.max_planets = max_planets
        self.max_fleets = max_fleets
        self.state_dim = state_dim
        self.action_dim = action_dim
        self.d_model = d_model

        # Use the actual P_network from SAC.py
        self.p_net = P_network(state_dim=state_dim, action_dim=action_dim,
                               max_planets=max_planets, max_fleets=max_fleets,
                               d_model=d_model, nhead=nhead, num_encoder_layers=num_layers,
                               dim_feedforward=dim_feedforward, dropout=dropout)
    
    def forward(self, state: Tensor) -> Tuple[Tensor, Tensor]:
        """
        Args:
            state: [batch_size, state_dim]
        
        Returns:
            mu: [batch_size, max_action, action_dim]
            log_std: [batch_size, max_action, action_dim]
        """
        batch_size = state.shape[0]
        mu, sigma = self.p_net(state)
        return mu, sigma

class V_NetworkAdapter(nn.Module):
    """
    Adapts the transformer V_network as features extractor for SB3.
    
    Input: [batch_size, state_dim]
    Output: [batch_size, features_dim]
    """
    
    def __init__(self, 
                 d_model=128, 
                 state_dim=14,
                 action_dim=8,
                 max_planets=40,
                 max_fleets=100, 
                 nhead=4, 
                 num_layers=3,  
                 dim_feedforward=128, 
                 dropout=0.1):
        super().__init__()
        self.max_planets = max_planets
        self.max_fleets = max_fleets    
        self.state_dim = state_dim
        self.action_dim = action_dim
        self.d_model = d_model
        self.num_state_tokens = state_dim // d_model
        
        # Use the actual V_network from SAC.py
        self.v_net = V_network(d_model=d_model, nhead=nhead, num_encoder_layers=num_layers,
                               dim_feedforward=dim_feedforward, dropout=dropout)
    
    def forward(self, state: Tensor) -> Tensor:
        """
        Args:
            state: [batch_size, state_dim]
        
        Returns:
            features: [batch_size, features_dim]
        """
        batch_size = state.shape[0]
        
        # Reshape to [batch_size, num_tokens, d_model]
        state_tokens = state.view(batch_size, self.num_state_tokens, self.d_model)
        
        # Forward through V_network (now handles batching natively)
        v_out = self.v_net(state_tokens)  # [batch_size]
        
        # Project to features_dim
        # Reshape v_out to [batch_size, 1] then to [batch_size, d_model] pattern
        features = self.output_proj(v_out.unsqueeze(-1).expand(-1, self.d_model))
        
        return features

# 3.  CUSTOM SAC POLICY WITH TRANSFORMER NETWORKS
# ─────────────────────────────────────────────

class TransformerSACPolicy(SACPolicy):
    """
    SACPolicy subclass using custom transformer-based Q_network and P_network
    (with V_network as features extractor) instead of default MLPs.
    """

    def __init__(
        self,
        observation_space: gym.Space,
        action_space: gym.Space,
        lr_schedule: Schedule,
        net_arch: Optional[List[int]] = None,
        activation_fn: Type[nn.Module] = nn.ReLU,
        use_sde: bool = False,
        log_std_init: float = -3,
        use_expln: bool = False,
        clip_mean: float = 2.0,
        features_extractor_kwargs: Optional[Dict[str, Any]] = None,
        normalize_images: bool = True,
        optimizer_class: Type[torch.optim.Optimizer] = torch.optim.Adam,
        optimizer_kwargs: Optional[Dict[str, Any]] = None,
        n_critics: int = 2,
        share_features_extractor: bool = False,
        d_model: int = 128,
        nhead: int = 4,
        num_encoder_layers: int = 2,
    ):
        if features_extractor_kwargs is None:
            features_extractor_kwargs = {
                "features_dim": 128,
                "d_model": d_model,
                "nhead": nhead,
                "num_encoder_layers": num_encoder_layers,
            }

        self.d_model = d_model
        self.nhead = nhead
        self.num_encoder_layers = num_encoder_layers

        super().__init__(
            observation_space=observation_space,
            action_space=action_space,
            lr_schedule=lr_schedule,
            net_arch=net_arch,
            activation_fn=activation_fn,
            use_sde=use_sde,
            log_std_init=log_std_init,
            use_expln=use_expln,
            clip_mean=clip_mean,
            features_extractor_class=features_extractor_class,
            features_extractor_kwargs=features_extractor_kwargs,
            normalize_images=normalize_images,
            optimizer_class=optimizer_class,
            optimizer_kwargs=optimizer_kwargs,
            n_critics=n_critics,
            share_features_extractor=share_features_extractor,
        )

    def make_actor(self, features_extractor: Optional[BaseFeaturesExtractor] = None) -> Actor:
        actor = super().make_actor(features_extractor)
        
        # Replace with custom transformer P_network adapter
        obs_dim = int(np.prod(self.observation_space.shape))
        action_dim = get_action_dim(self.action_space)
        
        actor.latent_pi = P_NetworkAdapter(
            state_dim=obs_dim,
            action_dim=action_dim,
            d_model=self.d_model,
            nhead=self.nhead,
            num_encoder_layers=self.num_encoder_layers,
        )
        
        # SB3 expects latent_dim_pi attribute for the output dimension
        actor.latent_dim_pi = action_dim * ((obs_dim // self.d_model) * (action_dim // self.d_model))
        
        # Override mu and log_std to be identity (adapter already outputs them)
        actor.mu = nn.Identity()
        actor.log_std = nn.Identity()

        return actor

    def make_critic(self, features_extractor: Optional[BaseFeaturesExtractor] = None):
        critic = super().make_critic(features_extractor)
        
        obs_dim = int(np.prod(self.observation_space.shape))
        action_dim = get_action_dim(self.action_space)
        
        # Replace Q-networks with custom transformer adapter
        critic.q_networks = nn.ModuleList([
            Q_NetworkAdapter(
                state_dim=obs_dim,
                action_dim=action_dim,
                d_model=self.d_model,
                nhead=self.nhead,
                num_encoder_layers=self.num_encoder_layers,
            )
            for _ in range(critic.n_critics)
        ])

        return critic
